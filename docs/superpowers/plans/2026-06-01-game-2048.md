// src/lib/game-2048.ts — 2048 游戏纯函数逻辑

export type Board = number[][] // 4×4
export type Direction = "up" | "down" | "left" | "right"

export interface MoveResult {
  board: Board
  score: number
  moved: boolean
}

const SIZE = 4

/** 创建空棋盘 */
function emptyBoard(): Board {
  return Array.from({ length: SIZE }, () => Array(SIZE).fill(0))
}

/** 随机在空位添加 2 (90%) 或 4 (10%) */
export function addRandomTile(board: Board): Board {
  const empty: [number, number][] = []
  for (let r = 0; r < SIZE; r++) {
    for (let c = 0; c < SIZE; c++) {
      if (board[r][c] === 0) empty.push([r, c])
    }
  }
  if (empty.length === 0) return board
  const [r, c] = empty[Math.floor(Math.random() * empty.length)]
  const next = structuredClone(board)
  next[r][c] = Math.random() < 0.9 ? 2 : 4
  return next
}

/** 创建初始棋盘：空棋盘 + 2 个 tile */
export function createBoard(): Board {
  let board = emptyBoard()
  board = addRandomTile(board)
  board = addRandomTile(board)
  return board
}

/** 沿单行向左合并，不修改原数组 */
function mergeRow(row: number[]): { merged: number[]; score: number } {
  // 去零
  const filtered = row.filter((v) => v !== 0)
  const merged: number[] = []
  let score = 0
  let i = 0
  while (i < filtered.length) {
    if (i + 1 < filtered.length && filtered[i] === filtered[i + 1]) {
      const val = filtered[i] * 2
      merged.push(val)
      score += val
      i += 2
    } else {
      merged.push(filtered[i])
      i += 1
    }
  }
  // 补齐到 SIZE
  while (merged.length < SIZE) merged.push(0)
  return { merged, score }
}

/** 逆时针旋转棋盘 90°（用于将不同方向统一为 left 操作） */
function rotate(board: Board): Board {
  const rotated = emptyBoard()
  for (let r = 0; r < SIZE; r++) {
    for (let c = 0; c < SIZE; c++) {
      rotated[SIZE - 1 - c][r] = board[r][c]
    }
  }
  return rotated
}

/** 执行一次移动，返回新棋盘和得分 */
export function move(board: Board, direction: Direction): MoveResult {
  // 将方向统一为 "left" 操作
  let rotated = board
  const rotations: Record<Direction, number> = {
    left: 0,
    up: 1,
    right: 2,
    down: 3,
  }

  for (let i = 0; i < rotations[direction]; i++) {
    rotated = rotate(rotated)
  }

  // 对每一行向左合并
  let totalScore = 0
  let moved = false
  const merged = emptyBoard()
  for (let r = 0; r < SIZE; r++) {
    const { merged: row, score } = mergeRow(rotated[r])
    merged[r] = row
    totalScore += score
    if (row.some((v, c) => v !== rotated[r][c])) moved = true
  }

  // 旋转回来
  let result = merged
  const reverseRotations = (4 - rotations[direction]) % 4
  for (let i = 0; i < reverseRotations; i++) {
    result = rotate(result)
  }

  return { board: result, score: totalScore, moved }
}

/** 检查棋盘是否有合法移动 */
export function canMove(board: Board): boolean {
  const directions: Direction[] = ["up", "down", "left", "right"]
  return directions.some((d) => move(board, d).moved)
}

/** 检查是否达到 2048 */
export function hasWon(board: Board): boolean {
  return board.some((row) => row.some((v) => v >= 2048))
}
```

// src/app/game/2048/page.tsx — 2048 游戏页面 ("use client")

"use client"

import { useState, useEffect, useCallback } from "react"
import type { Metadata } from "next"
import {
  createBoard,
  move,
  addRandomTile,
  canMove,
  hasWon,
} from "@/lib/game-2048"
import type { Board, Direction } from "@/lib/game-2048"

// localStorage 最高分 key
const BEST_KEY = "game-2048-best"

function getBestScore(): number {
  try {
    const val = localStorage.getItem(BEST_KEY)
    return val ? parseInt(val, 10) || 0 : 0
  } catch {
    return 0
  }
}

function setBestScore(score: number) {
  try {
    localStorage.setItem(BEST_KEY, String(score))
  } catch {
    // storage 不可用时静默降级
  }
}

// Tile 颜色映射
function tileStyle(value: number): {
  bg: string
  text: string
  fontSize: string
} {
  const map: Record<number, { bg: string; text: string }> = {
    2: { bg: "#EEE4DA", text: "#776E65" },
    4: { bg: "#EDE0C8", text: "#776E65" },
    8: { bg: "#F2B179", text: "#F9F6F2" },
    16: { bg: "#F59563", text: "#F9F6F2" },
    32: { bg: "#F67C5F", text: "#F9F6F2" },
    64: { bg: "#F65E3B", text: "#F9F6F2" },
    128: { bg: "#EDCF72", text: "#F9F6F2" },
    256: { bg: "#EDCC61", text: "#F9F6F2" },
    512: { bg: "#EDC850", text: "#F9F6F2" },
    1024: { bg: "#EDC53F", text: "#F9F6F2" },
    2048: { bg: "#EDC22E", text: "#F9F6F2" },
  }
  const entry = map[value]
  if (entry) return { ...entry, fontSize: value >= 128 ? "28px" : "36px" }
  // 4096+
  return { bg: "#3C3A32", text: "#F9F6F2", fontSize: "24px" }
}

export default function Game2048Page() {
  const [board, setBoard] = useState<Board>(createBoard)
  const [score, setScore] = useState(0)
  const [bestScore, setBestScoreState] = useState(getBestScore)
  const [gameOver, setGameOver] = useState(false)
  const [won, setWon] = useState(false)

  // 处理方向移动
  const handleMove = useCallback(
    (direction: Direction) => {
      if (gameOver) return

      setBoard((prev) => {
        const { board: newBoard, score: gain, moved } = move(prev, direction)
        if (!moved) return prev

        const withTile = addRandomTile(newBoard)
        setScore((s) => {
          const next = s + gain
          if (next > bestScore) {
            setBestScoreState(next)
            setBestScore(next)
          }
          return next
        })

        if (hasWon(withTile) && !won) setWon(true)

        // 延迟检查 game over（等状态更新后）
        setTimeout(() => {
          if (!canMove(withTile)) setGameOver(true)
        }, 0)

        return withTile
      })
    },
    [gameOver, bestScore, won],
  )

  // 重新开始
  const restart = useCallback(() => {
    setBoard(createBoard())
    setScore(0)
    setGameOver(false)
    setWon(false)
  }, [])

  // 键盘事件
  useEffect(() => {
    const keyMap: Record<string, Direction> = {
      ArrowUp: "up",
      ArrowDown: "down",
      ArrowLeft: "left",
      ArrowRight: "right",
    }

    const onKeyDown = (e: KeyboardEvent) => {
      const dir = keyMap[e.key]
      if (dir) {
        e.preventDefault()
        handleMove(dir)
      }
    }

    window.addEventListener("keydown", onKeyDown)
    return () => window.removeEventListener("keydown", onKeyDown)
  }, [handleMove])

  return (
    <main className="max-w-[1280px] mx-auto px-8 py-12 flex flex-col items-center">
      {/* 标题区 */}
      <div className="flex items-center gap-8 mb-6">
        <h1 className="text-3xl font-bold text-[#101828]">2048</h1>
        <div className="flex items-center gap-4">
          {/* 得分 */}
          <div className="bg-[#BBADA0] text-white rounded-[6px] px-4 py-2 text-center min-w-[80px]">
            <div className="text-xs uppercase tracking-wide">得分</div>
            <div className="text-lg font-bold">{score}</div>
          </div>
          {/* 最高分 */}
          <div className="bg-[#BBADA0] text-white rounded-[6px] px-4 py-2 text-center min-w-[80px]">
            <div className="text-xs uppercase tracking-wide">最高</div>
            <div className="text-lg font-bold">{bestScore}</div>
          </div>
        </div>
      </div>

      {/* 游戏说明 */}
      <p className="text-sm text-[#6A7282] mb-4">
        使用方向键 ← ↑ ↓ → 移动数字方块，相同数字合并相加，目标 2048！
      </p>

      {/* 棋盘区 */}
      <div className="relative">
        <div
          className="grid grid-cols-4 gap-2 p-3 rounded-[8px]"
          style={{ background: "#BBADA0" }}
        >
          {board.map((row, r) =>
            row.map((value, c) => {
              const style = tileStyle(value)
              return (
                <div
                  key={`${r}-${c}`}
                  className="size-[100px] rounded-[6px] flex items-center justify-center font-bold transition-all duration-100"
                  style={{
                    background: value === 0 ? "#CDC1B4" : style.bg,
                    color: style.text,
                    fontSize: style.fontSize,
                  }}
                >
                  {value !== 0 && value}
                </div>
              )
            }),
          )}
        </div>

        {/* Game Over 蒙层 */}
        {gameOver && (
          <div className="absolute inset-0 bg-black/50 rounded-[8px] flex flex-col items-center justify-center gap-3">
            <p className="text-white text-2xl font-bold">游戏结束</p>
            <button
              onClick={restart}
              className="bg-[#FB2C36] hover:bg-[#e0262f] text-white px-6 py-2 rounded-[6px] font-medium text-base transition-colors cursor-pointer"
            >
              再来一局
            </button>
          </div>
        )}

        {/* 胜利蒙层（允许继续） */}
        {won && !gameOver && (
          <div className="absolute inset-0 bg-black/30 rounded-[8px] flex flex-col items-center justify-center gap-3">
            <p className="text-white text-2xl font-bold">你赢了！</p>
            <p className="text-white/80 text-sm">继续游玩或开始新一局</p>
            <button
              onClick={() => setWon(false)}
              className="bg-[#FB2C36] hover:bg-[#e0262f] text-white px-6 py-2 rounded-[6px] font-medium text-base transition-colors cursor-pointer"
            >
              继续
            </button>
          </div>
        )}
      </div>

      {/* 重新开始按钮 */}
      <button
        onClick={restart}
        className="mt-6 bg-[#FB2C36] hover:bg-[#e0262f] text-white px-6 py-2 rounded-[6px] font-medium text-base transition-colors cursor-pointer"
      >
        重新开始
      </button>
    </main>
  )
}
```
